use anyhow::{Context, Result};
use libp2p::identity::{self, Keypair};
use libp2p::PeerId;
use serde::{Deserialize, Serialize};
use std::path::Path;
use tracing::info;

#[derive(Serialize, Deserialize)]
struct StoredIdentity {
    keypair_bytes: Vec<u8>,
}

pub struct NodeIdentity {
    keypair: Keypair,
    peer_id: PeerId,
}

impl NodeIdentity {
    pub fn generate() -> Self {
        let keypair = Keypair::generate_ed25519();
        let peer_id = keypair.public().to_peer_id();
        info!(%peer_id, "Generated new identity");
        Self { keypair, peer_id }
    }

    pub fn save_to_file(&self, path: &Path) -> Result<()> {
        let stored = StoredIdentity {
            keypair_bytes: self
                .keypair
                .to_protobuf_encoding()
                .context("Failed to encode keypair")?,
        };
        let json = serde_json::to_string_pretty(&stored)?;
        if let Some(parent) = path.parent() {
            std::fs::create_dir_all(parent)?;
        }
        std::fs::write(path, json)?;
        info!(?path, "Saved identity to file");
        Ok(())
    }

    pub fn load_from_file(path: &Path) -> Result<Self> {
        let json = std::fs::read_to_string(path).context("Failed to read identity file")?;
        let stored: StoredIdentity = serde_json::from_str(&json)?;
        let keypair =
            Keypair::from_protobuf_encoding(&stored.keypair_bytes).context("Invalid keypair")?;
        let peer_id = keypair.public().to_peer_id();
        info!(%peer_id, ?path, "Loaded identity from file");
        Ok(Self { keypair, peer_id })
    }

    pub fn load_or_generate(path: &Path) -> Result<Self> {
        if path.exists() {
            Self::load_from_file(path)
        } else {
            let identity = Self::generate();
            identity.save_to_file(path)?;
            Ok(identity)
        }
    }

    pub fn peer_id(&self) -> PeerId {
        self.peer_id
    }

    pub fn keypair(&self) -> &Keypair {
        &self.keypair
    }
}
```

            relay_addr: None,
            namespace: "clash-public".to_string(),

use crate::identity::NodeIdentity;
use anyhow::Result;
use libp2p::swarm::Swarm;
use libp2p::{autonat, identify, noise, ping, relay, rendezvous, yamux};
use tracing::info;

use super::behaviour::ClashBehaviour;

const PROTOCOL_VERSION: &str = "/clash/0.1.0";

pub async fn build_client_swarm(identity: &NodeIdentity) -> Result<Swarm<ClashBehaviour>> {
    let swarm = libp2p::SwarmBuilder::with_existing_identity(identity.keypair().clone())
        .with_tokio()
        .with_relay_client(noise::Config::new, yamux::Config::default)?
        .with_other_transport(|keypair| {
            libp2p::quic::tokio::Transport::new(libp2p::quic::Config::new(keypair))
        })?
        .with_behaviour(|key, relay_behaviour| {
            let peer_id = key.public().to_peer_id();
            ClashBehaviour {
                relay_client: relay_behaviour,
                identify: identify::Behaviour::new(identify::Config::new(
                    PROTOCOL_VERSION.to_string(),
                    key.public(),
                )),
                rendezvous_client: rendezvous::client::Behaviour::new(key.clone()),
                ping: ping::Behaviour::new(ping::Config::default()),
                autonat: autonat::v2::client::Behaviour::new(
                    libp2p::core::transport::dummy::DummyTransport::new(),
                    autonat::v2::client::Config::default(),
                ),
            }
        })?
        .with_swarm_config(|cfg| {
            cfg.with_idle_connection_timeout(std::time::Duration::from_secs(60))
        })
        .build();

    info!(peer_id = %swarm.local_peer_id(), "Built client swarm");
    Ok(swarm)
}
```

**Step 1: Implement relay main**

```rust
use anyhow::Result;
use clash_lib::config::default_data_dir;
use clash_lib::identity::NodeIdentity;
use clash_lib::network::behaviour::RelayBehaviourEvent;
use clash_lib::network::swarm::build_relay_swarm;
use clap::Parser;
use libp2p::swarm::SwarmEvent;
use libp2p::Multiaddr;
use tokio::signal;
use tracing::{info, warn};
use tracing_subscriber::EnvFilter;

#[derive(Parser)]
#[command(name = "clash-relay", about = "Clash P2P relay/rendezvous server")]
struct Cli {
    /// Listen address
    #[arg(short, long, default_value = "/ip4/0.0.0.0/udp/4001/quic-v1")]
    listen: String,
}

#[tokio::main]
async fn main() -> Result<()> {
    tracing_subscriber::fmt()
        .with_env_filter(
            EnvFilter::try_from_default_env().unwrap_or_else(|_| EnvFilter::new("info")),
        )
        .init();

    let cli = Cli::parse();

    let data_dir = default_data_dir().join("relay");
    let identity_path = data_dir.join("identity.json");
    let identity = NodeIdentity::load_or_generate(&identity_path)?;

    info!(peer_id = %identity.peer_id(), "Starting relay server");

    let mut swarm = build_relay_swarm(&identity).await?;

    let listen_addr: Multiaddr = cli.listen.parse()?;
    swarm.listen_on(listen_addr)?;

    loop {
        tokio::select! {
            event = swarm.select_next_some() => {
                match event {
                    SwarmEvent::NewListenAddr { address, .. } => {
                        info!(%address, peer_id = %swarm.local_peer_id(),
                            "Relay listening on {address}/p2p/{}", swarm.local_peer_id());
                    }
                    SwarmEvent::ConnectionEstablished { peer_id, .. } => {
                        info!(%peer_id, "Peer connected");
                    }
                    SwarmEvent::ConnectionClosed { peer_id, .. } => {
                        info!(%peer_id, "Peer disconnected");
                    }
                    SwarmEvent::Behaviour(event) => {
                        match event {
                            RelayBehaviourEvent::Rendezvous(e) => info!(?e, "Rendezvous event"),
                            RelayBehaviourEvent::Relay(e) => info!(?e, "Relay event"),
                            _ => {}
                        }
                    }
                    _ => {}
                }
            }
            _ = signal::ctrl_c() => {
                info!("Shutting down relay server");
                break;
            }
        }
    }

    Ok(())
}
```

### Task 6: Implement clash-node Binary with Rendezvous Discovery

**Files:**
- Modify: `clash-node/src/main.rs`

The node binary connects to the relay, registers via Rendezvous, and discovers peers.

**Step 1: Implement node main**

```rust
use anyhow::Result;
use clash_lib::config::default_data_dir;
use clash_lib::identity::NodeIdentity;
use clash_lib::network::behaviour::ClashBehaviourEvent;
use clash_lib::network::swarm::build_client_swarm;
use clap::Parser;
use libp2p::rendezvous;
use libp2p::swarm::SwarmEvent;
use libp2p::{Multiaddr, PeerId};
use std::time::Duration;
use tokio::{signal, time};
use tracing::{info, warn};
use tracing_subscriber::EnvFilter;

#[derive(Parser)]
#[command(name = "clash-node", about = "Clash P2P proxy node")]
struct Cli {
    /// Relay server multiaddr (e.g. /ip4/127.0.0.1/udp/4001/quic-v1/p2p/<PEER_ID>)
    #[arg(short, long)]
    relay: String,

    /// Namespace for rendezvous
    #[arg(short, long, default_value = "clash-public")]
    namespace: String,

    /// Run as exit node
    #[arg(short, long, default_value_t = false)]
    exit_node: bool,
}

#[tokio::main]
async fn main() -> Result<()> {
    tracing_subscriber::fmt()
        .with_env_filter(
            EnvFilter::try_from_default_env().unwrap_or_else(|_| EnvFilter::new("info")),
        )
        .init();

    let cli = Cli::parse();

    let data_dir = default_data_dir().join("node");
    let identity_path = data_dir.join("identity.json");
    let identity = NodeIdentity::load_or_generate(&identity_path)?;

    info!(peer_id = %identity.peer_id(), "Starting clash node");

    let mut swarm = build_client_swarm(&identity).await?;

    // Listen on random port
    swarm.listen_on("/ip4/0.0.0.0/udp/0/quic-v1".parse()?)?;

    // Parse relay address and connect
    let relay_addr: Multiaddr = cli.relay.parse()?;
    let relay_peer_id = relay_addr
        .iter()
        .find_map(|p| match p {
            libp2p::multiaddr::Protocol::P2p(peer_id) => Some(peer_id),
            _ => None,
        })
        .expect("Relay address must contain /p2p/<peer_id>");

    swarm.dial(relay_addr.clone())?;
    info!(%relay_peer_id, "Dialing relay server");

    let namespace = cli.namespace.clone();
    let mut registered = false;
    let mut discover_interval = time::interval(Duration::from_secs(30));

    loop {
        tokio::select! {
            event = swarm.select_next_some() => {
                match event {
                    SwarmEvent::NewListenAddr { address, .. } => {
                        info!(%address, "Listening");
                    }
                    SwarmEvent::ConnectionEstablished { peer_id, .. } => {
                        info!(%peer_id, "Connected to peer");
                        if peer_id == relay_peer_id && !registered {
                            // Register with rendezvous
                            swarm.behaviour_mut().rendezvous_client.register(
                                rendezvous::Namespace::from_static(&namespace),
                                relay_peer_id,
                                None,
                            )?;
                            info!("Registering with rendezvous");
                        }
                    }
                    SwarmEvent::Behaviour(ClashBehaviourEvent::RendezvousClient(event)) => {
                        match event {
                            rendezvous::client::Event::Registered { namespace, .. } => {
                                info!(?namespace, "Registered with rendezvous");
                                registered = true;
                                // Discover other peers
                                swarm.behaviour_mut().rendezvous_client.discover(
                                    Some(rendezvous::Namespace::from_static(&namespace)),
                                    None,
                                    None,
                                    relay_peer_id,
                                );
                            }
                            rendezvous::client::Event::Discovered { registrations, .. } => {
                                for reg in registrations {
                                    let peer = reg.record.peer_id();
                                    if peer != *swarm.local_peer_id() {
                                        info!(%peer, "Discovered peer");
                                        // Try to dial discovered peer
                                        if let Err(e) = swarm.dial(peer) {
                                            warn!(%peer, %e, "Failed to dial peer");
                                        }
                                    }
                                }
                            }
                            other => info!(?other, "Rendezvous event"),
                        }
                    }
                    SwarmEvent::Behaviour(ClashBehaviourEvent::Ping(event)) => {
                        info!(?event, "Ping");
                    }
                    SwarmEvent::ConnectionClosed { peer_id, .. } => {
                        info!(%peer_id, "Disconnected");
                    }
                    _ => {}
                }
            }
            _ = discover_interval.tick() => {
                if registered {
                    swarm.behaviour_mut().rendezvous_client.discover(
                        Some(rendezvous::Namespace::from_static(&namespace)),
                        None,
                        None,
                        relay_peer_id,
                    );
                }
            }
            _ = signal::ctrl_c() => {
                info!("Shutting down");
                break;
            }
        }
    }

    Ok(())
}
```
