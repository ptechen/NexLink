# P4: Tauri + Leptos Desktop GUI

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a complete desktop GUI with Tauri v2 + Leptos CSR + Tailwind CSS, featuring dark tech theme, three pages (Nodes, Traffic, Settings), and real-time status updates.

**Architecture:** Tauri v2 backend holds the libp2p swarm in a background tokio task, communicates via mpsc channels. Leptos CSR frontend compiled to WASM via Trunk, calls Tauri commands via `wasm-bindgen`. Backend pushes real-time events (peer discovery, traffic, NAT status) to frontend via Tauri event system.

**Tech Stack:** Tauri 2.x, Leptos 0.8 (CSR), leptos_router 0.8, Tailwind CSS v4, Trunk, wasm-bindgen, serde-wasm-bindgen

**Context:** The Tauri backend needs to manage the libp2p swarm, expose IPC commands for the frontend, and emit events. We use `tokio::sync::mpsc` for command passing and `Arc<RwLock<SharedState>>` for shared readable state.

**Step 1: Create state.rs**

Create `clash-app/src-tauri/src/state.rs`:

```rust
use serde::{Deserialize, Serialize};
use std::sync::Arc;
use tokio::sync::{mpsc, RwLock};

#[derive(Debug, Clone, Serialize, Deserialize)]
#[serde(rename_all = "camelCase")]
pub struct PeerInfo {
    pub peer_id: String,
    pub addrs: Vec<String>,
    pub is_exit_node: bool,
    pub latency_ms: Option<u64>,
    pub connected: bool,
}

#[derive(Debug, Clone, Default, Serialize, Deserialize)]
#[serde(rename_all = "camelCase")]
pub struct TrafficStats {
    pub bytes_sent: u64,
    pub bytes_received: u64,
    pub upload_speed: u64,
    pub download_speed: u64,
    pub active_connections: u32,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
#[serde(rename_all = "camelCase")]
pub struct ProxyStatus {
    pub running: bool,
    pub socks5_port: u16,
    pub http_port: u16,
}

#[derive(Debug, Clone, Default, Serialize, Deserialize)]
#[serde(rename_all = "camelCase")]
pub struct SharedState {
    pub peer_id: String,
    pub proxy_status: Option<ProxyStatus>,
    pub connected_peer: Option<String>,
    pub discovered_peers: Vec<PeerInfo>,
    pub nat_status: String,
    pub traffic: TrafficStats,
    pub relay_addr: String,
    pub namespace: String,
    pub data_dir: String,
}

#[derive(Debug)]
pub enum AppCommand {
    StartProxy {
        socks5_port: u16,
        http_port: u16,
    },
    StopProxy,
    ConnectNode {
        peer_id: String,
    },
    DisconnectNode,
    RefreshNodes,
    UpdateConfig {
        relay_addr: Option<String>,
        namespace: Option<String>,
    },
}

pub struct AppState {
    pub cmd_tx: mpsc::Sender<AppCommand>,
    pub shared: Arc<RwLock<SharedState>>,
}
```

```rust
mod commands;
mod state;

use state::{AppCommand, AppState, SharedState};
use std::sync::Arc;
use tokio::sync::{mpsc, RwLock};

#[cfg_attr(mobile, tauri::mobile_entry_point)]
pub fn run() {
    tracing_subscriber::fmt()
        .with_env_filter("clash=debug,info")
        .init();

    let (cmd_tx, _cmd_rx) = mpsc::channel::<AppCommand>(32);
    let shared = Arc::new(RwLock::new(SharedState {
        peer_id: "not-connected".to_string(),
        nat_status: "Unknown".to_string(),
        relay_addr: String::new(),
        namespace: "clash-public".to_string(),
        data_dir: clash_lib::config::default_data_dir()
            .to_string_lossy()
            .to_string(),
        ..Default::default()
    }));

    let app_state = AppState {
        cmd_tx,
        shared: shared.clone(),
    };

    tauri::Builder::default()
        .manage(app_state)
        .invoke_handler(tauri::generate_handler![
            commands::get_identity,
            commands::get_status,
            commands::list_nodes,
            commands::connect_node,
            commands::disconnect_node,
            commands::start_proxy,
            commands::stop_proxy,
            commands::get_traffic,
            commands::get_proxy_status,
            commands::update_config,
        ])
        .run(tauri::generate_context!())
        .expect("error while running tauri application");
}
```

**Step 2: Create api.rs**

Create `clash-app/src/api.rs`:

```rust
use serde::Serialize;
use wasm_bindgen::prelude::*;

use crate::types::{AppStatus, PeerInfo, ProxyStatus, TrafficStats};

#[wasm_bindgen]
extern "C" {
    #[wasm_bindgen(catch, js_namespace = ["window", "__TAURI__", "core"])]
    async fn invoke(cmd: &str, args: JsValue) -> Result<JsValue, JsValue>;
}

#[wasm_bindgen]
extern "C" {
    #[wasm_bindgen(catch, js_namespace = ["window", "__TAURI__", "event"])]
    async fn listen(event: &str, handler: &Closure<dyn FnMut(JsValue)>) -> Result<JsValue, JsValue>;
}

async fn call<T: serde::de::DeserializeOwned>(cmd: &str, args: JsValue) -> Result<T, String> {
    let result = invoke(cmd, args)
        .await
        .map_err(|e| e.as_string().unwrap_or_else(|| "Unknown error".to_string()))?;
    serde_wasm_bindgen::from_value(result).map_err(|e| e.to_string())
}

async fn call_void(cmd: &str, args: JsValue) -> Result<(), String> {
    invoke(cmd, args)
        .await
        .map_err(|e| e.as_string().unwrap_or_else(|| "Unknown error".to_string()))?;
    Ok(())
}

fn no_args() -> JsValue {
    serde_wasm_bindgen::to_value(&serde_json::json!({})).unwrap()
}

pub async fn get_identity() -> Result<String, String> {
    call::<String>("get_identity", no_args()).await
}

pub async fn get_status() -> Result<AppStatus, String> {
    call::<AppStatus>("get_status", no_args()).await
}

pub async fn list_nodes() -> Result<Vec<PeerInfo>, String> {
    call::<Vec<PeerInfo>>("list_nodes", no_args()).await
}

pub async fn connect_node(peer_id: &str) -> Result<(), String> {
    #[derive(Serialize)]
    #[serde(rename_all = "camelCase")]
    struct Args<'a> {
        peer_id: &'a str,
    }
    let args = serde_wasm_bindgen::to_value(&Args { peer_id }).map_err(|e| e.to_string())?;
    call_void("connect_node", args).await
}

pub async fn disconnect_node() -> Result<(), String> {
    call_void("disconnect_node", no_args()).await
}

pub async fn start_proxy(socks5_port: u16, http_port: u16) -> Result<(), String> {
    #[derive(Serialize)]
    #[serde(rename_all = "camelCase")]
    struct Args {
        socks5_port: u16,
        http_port: u16,
    }
    let args =
        serde_wasm_bindgen::to_value(&Args { socks5_port, http_port }).map_err(|e| e.to_string())?;
    call_void("start_proxy", args).await
}

pub async fn stop_proxy() -> Result<(), String> {
    call_void("stop_proxy", no_args()).await
}

pub async fn get_traffic() -> Result<TrafficStats, String> {
    call::<TrafficStats>("get_traffic", no_args()).await
}

pub async fn get_proxy_status() -> Result<Option<ProxyStatus>, String> {
    call::<Option<ProxyStatus>>("get_proxy_status", no_args()).await
}

pub async fn update_config(relay_addr: Option<&str>, namespace: Option<&str>) -> Result<(), String> {
    #[derive(Serialize)]
    #[serde(rename_all = "camelCase")]
    struct Args<'a> {
        relay_addr: Option<&'a str>,
        namespace: Option<&'a str>,
    }

        relay_addr,
        namespace,
    })
```

```rust
fn format_bytes(bytes: u64) -> String {
    if bytes == 0 {
        return "0 B".to_string();
    }
    let units = ["B", "KB", "MB", "GB", "TB"];
    let mut size = bytes as f64;
    let mut unit_idx = 0;
    while size >= 1024.0 && unit_idx < units.len() - 1 {
        size /= 1024.0;
        unit_idx += 1;
    }
    if unit_idx == 0 {
        format!("{} B", bytes)
    } else {
        format!("{:.1} {}", size, units[unit_idx])
    }
}

fn format_speed(bytes_per_sec: u64) -> String {
    format!("{}/s", format_bytes(bytes_per_sec))
}
```

```rust
fn format_speed(bytes_per_sec: u64) -> String {
    if bytes_per_sec == 0 {
        return "0 B/s".to_string();
    }
    let units = ["B/s", "KB/s", "MB/s", "GB/s"];
    let mut size = bytes_per_sec as f64;
    let mut unit_idx = 0;
    while size >= 1024.0 && unit_idx < units.len() - 1 {
        size /= 1024.0;
        unit_idx += 1;
    }
    if unit_idx == 0 {
        format!("{} B/s", bytes_per_sec)
    } else {
        format!("{:.1} {}", size, units[unit_idx])
    }
}
```

**Context:** This is the critical integration task. We spawn a background tokio task that runs the libp2p swarm event loop, processes `AppCommand`s from the frontend, and emits Tauri events for real-time updates. This reuses `clash-lib` components (identity, network, proxy).

**Step 1: Create swarm_task.rs**

Create `clash-app/src-tauri/src/swarm_task.rs`:

```rust
use crate::state::{AppCommand, PeerInfo, ProxyStatus, SharedState, TrafficStats};
use anyhow::Result;
use clash_lib::identity::NodeIdentity;
use clash_lib::network::behaviour::ClashBehaviourEvent;
use clash_lib::network::swarm::build_client_swarm;
use clash_lib::proxy;
use libp2p::swarm::SwarmEvent;
use libp2p::{autonat, relay, rendezvous, Multiaddr, PeerId};
use libp2p_stream as stream;
use std::sync::Arc;
use tauri::{AppHandle, Emitter};
use tokio::sync::{mpsc, RwLock};
use tokio::time::{interval, Duration};
use tracing::{info, warn, debug};

pub async fn run_swarm_task(
    app: AppHandle,
    mut cmd_rx: mpsc::Receiver<AppCommand>,
    shared: Arc<RwLock<SharedState>>,
    data_dir: String,
) -> Result<()> {
    let identity_path = std::path::Path::new(&data_dir).join("identity.json");
    let identity = NodeIdentity::load_or_generate(&identity_path)?;
    let peer_id = identity.peer_id();

    // Update shared state with peer ID
    {
        let mut state = shared.write().await;
        state.peer_id = peer_id.to_string();
        state.data_dir = data_dir.clone();
    }

    let mut swarm = build_client_swarm(&identity).await?;
    let stream_control = swarm.behaviour().stream.new_control();

    // Listen on random QUIC port
    swarm.listen_on("/ip4/0.0.0.0/udp/0/quic-v1".parse()?)?;

    let mut discover_tick = interval(Duration::from_secs(30));
    let mut relay_peer_id: Option<PeerId> = None;
    let mut relay_addr: Option<Multiaddr> = None;
    let mut connected_exit_peer: Option<PeerId> = None;
    let mut registered = false;
    let mut proxy_handles: Vec<tokio::task::JoinHandle<()>> = Vec::new();

    // Handle incoming proxy streams (exit node mode is off by default in GUI)
    // The GUI acts as a client node only

    info!(%peer_id, "Swarm task started");
    let _ = app.emit("swarm-ready", peer_id.to_string());

    loop {
        tokio::select! {
            event = swarm.select_next_some() => {
                match event {
                    SwarmEvent::NewListenAddr { address, .. } => {
                        debug!(%address, "Listening on");
                    }
                    SwarmEvent::ConnectionEstablished { peer_id: remote, .. } => {
                        debug!(%remote, "Connection established");

                        // If connected to relay and not registered, register with rendezvous
                        if Some(remote) == relay_peer_id && !registered {
                            if let Some(ref addr) = relay_addr {
                                let namespace = shared.read().await.namespace.clone();
                                swarm.behaviour_mut().rendezvous_client.register(
                                    rendezvous::Namespace::from_static("clash-public"),
                                    remote,
                                    None,
                                );
                                registered = true;
                            }
                        }
                    }
                    SwarmEvent::Behaviour(ClashBehaviourEvent::Identify(identify::Event::Received { peer_id: remote, info, .. })) => {
                        debug!(%remote, "Identified peer");
                        // After identify completes with relay, do rendezvous discover
                        if Some(remote) == relay_peer_id {
                            let namespace = shared.read().await.namespace.clone();
                            swarm.behaviour_mut().rendezvous_client.discover(
                                Some(rendezvous::Namespace::from_static("clash-public")),
                                None,
                                None,
                                remote,
                            );
                        }
                    }
                    SwarmEvent::Behaviour(ClashBehaviourEvent::RendezvousClient(
                        rendezvous::client::Event::Discovered { registrations, .. }
                    )) => {
                        let mut state = shared.write().await;
                        for reg in registrations {
                            let discovered_peer = reg.record.peer_id();
                            if discovered_peer == peer_id {
                                continue; // skip self
                            }
                            let addrs: Vec<String> = reg.record.addresses().iter()
                                .map(|a| a.to_string())
                                .collect();

                            // Check if already known
                            if !state.discovered_peers.iter().any(|p| p.peer_id == discovered_peer.to_string()) {
                                let peer_info = PeerInfo {
                                    peer_id: discovered_peer.to_string(),
                                    addrs,
                                    is_exit_node: true, // assume discovered peers are exit nodes
                                    latency_ms: None,
                                    connected: false,
                                };
                                state.discovered_peers.push(peer_info.clone());
                                let _ = app.emit("peer-discovered", &peer_info);
                            }

                            // Dial the peer
                            let _ = swarm.dial(discovered_peer);
                        }
                    }
                    SwarmEvent::Behaviour(ClashBehaviourEvent::Autonat(autonat::Event::StatusChanged { new, .. })) => {
                        let nat_str = match &new {
                            autonat::NatStatus::Public(_) => "Public",
                            autonat::NatStatus::Private => "Private",
                            autonat::NatStatus::Unknown => "Unknown",
                        };
                        {
                            let mut state = shared.write().await;
                            state.nat_status = nat_str.to_string();
                        }
                        let _ = app.emit("nat-status", nat_str);
                    }
                    SwarmEvent::Behaviour(ClashBehaviourEvent::RelayClient(
                        relay::client::Event::ReservationReqAccepted { .. }
                    )) => {
                        info!("Relay reservation accepted");
                    }
                    SwarmEvent::OutgoingConnectionError { peer_id: Some(remote), error, .. } => {
                        warn!(%remote, "Connection failed: {error}");
                        // Retry via relay circuit if available
                        if let (Some(ref r_addr), Some(r_peer)) = (&relay_addr, relay_peer_id) {
                            if remote != r_peer {
                                let circuit_addr: Multiaddr = format!(
                                    "{}/p2p-circuit/p2p/{}", r_addr, remote
                                ).parse().unwrap_or_else(|_| return);
                                let _ = swarm.dial(
                                    libp2p::swarm::dial_opts::DialOpts::peer_id(remote)
                                        .addresses(vec![circuit_addr])
                                        .condition(libp2p::swarm::dial_opts::PeerCondition::Always)
                                        .build()
                                );
                            }
                        }
                    }
                    _ => {}
                }
            }

            _ = discover_tick.tick() => {
                if let Some(relay) = relay_peer_id {
                    swarm.behaviour_mut().rendezvous_client.discover(
                        Some(rendezvous::Namespace::from_static("clash-public")),
                        None,
                        None,
                        relay,
                    );
                }
            }

            Some(cmd) = cmd_rx.recv() => {
                match cmd {
                    AppCommand::StartProxy { socks5_port, http_port } => {
                        if let Some(exit_peer) = connected_exit_peer {
                            let control = stream_control.clone();
                            let h1 = tokio::spawn(async move {
                                if let Err(e) = proxy::socks5::start_socks5_proxy(socks5_port, exit_peer, control).await {
                                    warn!("SOCKS5 proxy error: {e}");
                                }
                            });
                            let control2 = stream_control.clone();
                            let h2 = tokio::spawn(async move {
                                if let Err(e) = proxy::http_connect::start_http_proxy(http_port, exit_peer, control2).await {
                                    warn!("HTTP proxy error: {e}");
                                }
                            });
                            proxy_handles.push(h1);
                            proxy_handles.push(h2);

                            let status = ProxyStatus { running: true, socks5_port, http_port };
                            shared.write().await.proxy_status = Some(status.clone());
                            let _ = app.emit("proxy-status", &status);
                        } else {
                            warn!("No exit peer connected, cannot start proxy");
                        }
                    }
                    AppCommand::StopProxy => {
                        for h in proxy_handles.drain(..) {
                            h.abort();
                        }
                        let status = ProxyStatus { running: false, socks5_port: 1080, http_port: 8080 };
                        shared.write().await.proxy_status = Some(status.clone());
                        let _ = app.emit("proxy-status", &status);
                    }
                    AppCommand::ConnectNode { peer_id: target } => {
                        if let Ok(pid) = target.parse::<PeerId>() {
                            let _ = swarm.dial(pid);
                            connected_exit_peer = Some(pid);
                            {
                                let mut state = shared.write().await;
                                state.connected_peer = Some(target.clone());
                                for peer in state.discovered_peers.iter_mut() {
                                    peer.connected = peer.peer_id == target;
                                }
                            }
                            let _ = app.emit("peer-connected", &target);
                        }
                    }
                    AppCommand::DisconnectNode => {
                        if let Some(exit_peer) = connected_exit_peer.take() {
                            let _ = swarm.disconnect_peer_id(exit_peer);
                            let mut state = shared.write().await;
                            state.connected_peer = None;
                            for peer in state.discovered_peers.iter_mut() {
                                peer.connected = false;
                            }
                        }
                        let _ = app.emit("peer-disconnected", ());
                    }
                    AppCommand::RefreshNodes => {
                        if let Some(relay) = relay_peer_id {
                            swarm.behaviour_mut().rendezvous_client.discover(
                                Some(rendezvous::Namespace::from_static("clash-public")),
                                None,
                                None,
                                relay,
                            );
                        }
                    }
                    AppCommand::UpdateConfig { relay_addr: new_relay, namespace: new_ns } => {
                        let mut state = shared.write().await;
                        if let Some(addr) = new_relay {
                            state.relay_addr = addr.clone();
                            // Parse and dial new relay
                            if let Ok(maddr) = addr.parse::<Multiaddr>() {
                                // Extract peer ID from multiaddr
                                if let Some(libp2p::multiaddr::Protocol::P2p(pid)) = maddr.iter().last() {
                                    relay_peer_id = Some(pid);
                                    relay_addr = Some(maddr.clone());
                                    let _ = swarm.dial(maddr);
                                    registered = false;
                                }
                            }
                        }
                        if let Some(ns) = new_ns {
                            state.namespace = ns;
                        }
                    }
                }
            }
        }
    }
}
```

```rust
use std::sync::Arc;
use tauri::Manager;
use tokio::sync::{mpsc, RwLock};

#[cfg_attr(mobile, tauri::mobile_entry_point)]
pub fn run() {
    tracing_subscriber::fmt()
        .with_env_filter("clash=debug,info")
        .init();

    let (cmd_tx, cmd_rx) = mpsc::channel::<AppCommand>(32);
    let shared = Arc::new(RwLock::new(SharedState {
        peer_id: "initializing...".to_string(),
        nat_status: "Unknown".to_string(),
        relay_addr: String::new(),
        namespace: "clash-public".to_string(),
        data_dir: clash_lib::config::default_data_dir()
            .to_string_lossy()
            .to_string(),
        ..Default::default()
    }));

    let app_state = AppState {
        cmd_tx,
        shared: shared.clone(),
    };

    tauri::Builder::default()
        .manage(app_state)
        .setup(move |app| {
            let handle = app.handle().clone();
            let shared_clone = shared.clone();
            let data_dir = clash_lib::config::default_data_dir()
                .to_string_lossy()
                .to_string();

            tauri::async_runtime::spawn(async move {
                if let Err(e) = swarm_task::run_swarm_task(
                    handle,
                    cmd_rx,
                    shared_clone,
                    data_dir,
                ).await {
                    tracing::error!("Swarm task error: {e}");
                }
            });

            Ok(())
        })
        .invoke_handler(tauri::generate_handler![
```
